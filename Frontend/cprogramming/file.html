File Operations in C

File operations in C are essential for working with files (reading, writing, appending) within programs. These operations are facilitated through standard library functions, and files are opened using specific modes that define the kind of operations that can be performed.

1. File Handling Functions in C

Here are the core functions used for file handling in C:

  • fopen(): Opens a file with a specific mode.
  • fclose(): Closes an opened file.
  • fread(): Reads data from a file.
  • fwrite(): Writes data to a file.
  • fseek(): Moves the file pointer to a specific position.
  • ftell(): Returns the current position of the file pointer.
  • fprintf(): Writes formatted data to a file (similar to printf).
  • fscanf(): Reads formatted input from a file (similar to scanf).
  • feof(): Checks if the end-of-file (EOF) has been reached.
  • ferror(): Checks if an error occurred during a file operation.

2. File Opening Modes in C

File opening modes specify the actions allowed on the file when it is opened. Below are the common file modes used in C:

  1. "r" (Read Mode): Opens a file for reading only. The file must exist.
    2. FILE *fp = fopen("file.txt", "r");
if (fp == NULL) {
    printf("File not found\n");
}
  2. "w" (Write Mode): Opens a file for writing. If the file exists, it truncates the file (empties it). If it does not exist, a new file is created.
    3. FILE *fp = fopen("file.txt", "w");
if (fp != NULL) {
    fprintf(fp, "Hello, World!");
    fclose(fp);
}
  3. "a" (Append Mode): Opens a file for appending. Data will be written at the end of the file, without truncating it.
    4. FILE *fp = fopen("file.txt", "a");
if (fp != NULL) {
    fprintf(fp, "New line of text\n");
    fclose(fp);
}
  4. "r+" (Read and Write Mode): Opens a file for both reading and writing. The file must exist.
    5. FILE *fp = fopen("file.txt", "r+");
if (fp != NULL) {
    char ch = fgetc(fp);  // Reading a character
    fseek(fp, 0, SEEK_END);  // Move pointer to the end
    fprintf(fp, "New data\n");
    fclose(fp);
}
  5. "w+" (Write and Read Mode): Opens a file for both reading and writing. If the file exists, it is truncated. If it does not exist, a new file is created.
    6. FILE *fp = fopen("file.txt", "w+");
if (fp != NULL) {
    fprintf(fp, "Hello again!");
    fseek(fp, 0, SEEK_SET);
    char ch = fgetc(fp);  // Read the first character
    printf("First character: %c\n", ch);
    fclose(fp);
}
  6. "a+" (Append and Read Mode): Opens a file for both reading and appending. The file pointer is placed at the end of the file for appending, but you can also read the file from the beginning or anywhere.
    7. FILE *fp = fopen("file.txt", "a+");
if (fp != NULL) {
    fprintf(fp, "Appending data\n");
    fseek(fp, 0, SEEK_SET);  // Go to the start to read
    char ch = fgetc(fp);  // Read the first character
    printf("First character: %c\n", ch);
    fclose(fp);
}
  7. "b" (Binary Mode): This mode is used for handling binary files. It can be combined with other modes ("rb", "wb", "ab", etc.).
    8. FILE *fp = fopen("file.bin", "wb");
if (fp != NULL) {
    int data = 123;
    fwrite(&data, sizeof(data), 1, fp);  // Writing binary data
    fclose(fp);
}

3. File Handling Example (Read, Write, Append)

#include <stdio.h>
int main() {
    FILE *fp;

    // Write data to the file
    fp = fopen("example.txt", "w");
    if (fp != NULL) {
        fprintf(fp, "This is a test file.\n");
        fclose(fp);
    } else {
        printf("Error opening file for writing.\n");
    }

    // Read data from the file
    fp = fopen("example.txt", "r");
    if (fp != NULL) {
        char ch;
        while ((ch = fgetc(fp)) != EOF) {
            putchar(ch);  // Print the data to the console
        }
        fclose(fp);
    } else {
        printf("Error opening file for reading.\n");
    }

    // Append data to the file
    fp = fopen("example.txt", "a");
    if (fp != NULL) {
        fprintf(fp, "Appended text.\n");
        fclose(fp);
    } else {
        printf("Error opening file for appending.\n");
    }

    return 0;
}

4. Advantages and Disadvantages of File Handling

Advantages:

  • Persistence: Files allow storing data persistently, making it accessible between program executions.
  • Data Management: Files can be used to manage large data sets in external storage.
  • Flexibility: Files can be read and written in various formats (binary or text).
  • Multiple Operations: Allows operations like search, update, and delete on large data sets.

Disadvantages:

  • Performance: File I/O is generally slower than working with memory.
  • Error Handling: File handling requires proper error checking, especially for file opening and closing.
  • Concurrency: If multiple processes or threads try to access the file simultaneously, data corruption can occur.
  • Security: Sensitive data can be vulnerable if the file is not securely handled.

1.Write a program to copy content from one file to another 

#include <stdio.h>
#include <stdlib.h>
int main() {
    FILE *sourceFile, *destinationFile;
    char ch;

    sourceFile = fopen("source.txt", "r");
    if (sourceFile == NULL) {
        printf("Error: Could not open source file.\n");
        return 1;
    }

    destinationFile = fopen("destination.txt", "w");
    if (destinationFile == NULL) {
        printf("Error: Could not open destination file.\n");
        fclose(sourceFile);
        return 1;
    }

    while ((ch = fgetc(sourceFile)) != EOF) {
        fputc(ch, destinationFile);
    }

    printf("File content copied successfully from source.txt to destination.txt.\n");

    fclose(sourceFile);
    fclose(destinationFile);

    return 0;
}

3.Write a Program to Write User Input to a File

#include <stdio.h>
#include <stdlib.h>
int main() {
    FILE *file;
    char filename[100];
    char input[1000];

    printf("Enter the filename to save data: ");
    scanf("%s", filename);

    file = fopen(filename, "w");

    if (file == NULL) {
        printf("Error: Could not open the file %s\n", filename);
        exit(1);
    }

    printf("Enter the text to write to the file (press Enter to finish):\n");
    getchar(); // To consume the newline character left by previous scanf
    fgets(input, sizeof(input), stdin);

    fprintf(file, "%s", input);

    fclose(file);

    printf("Data has been written to the file %s\n", filename);

    return 0;
}

4.Count the Number of Words in a File

#include <stdio.h>
#include <ctype.h>

int main() {
    FILE *file;
    char filename[100];
    char ch;
    int inWord = 0;
    int wordCount = 0;

    printf("Enter the filename to count words: ");
    scanf("%s", filename);

    file = fopen(filename, "r");

    if (file == NULL) {
        printf("Error: Could not open the file %s\n", filename);
        return 1;
    }

    while ((ch = fgetc(file)) != EOF) {
        if (isalpha(ch) || ch == '\'' || ch == '-') {
            if (!inWord) {
                inWord = 1;
                wordCount++;
            }
        } else {
            inWord = 0;
        }
    }

    fclose(file);

    printf("The file contains %d words.\n", wordCount);

    return 0;
}

5.Program to Merge Two Files into One

#include <stdio.h>
#include <stdlib.h>
int main() {
    FILE *file1, *file2, *mergedFile;
    char filename1[100], filename2[100], mergedFilename[100];
    char ch;

    printf("Enter the first file name: ");
    scanf("%s", filename1);

    printf("Enter the second file name: ");
    scanf("%s", filename2);

    printf("Enter the name for the merged file: ");
    scanf("%s", mergedFilename);

    file1 = fopen(filename1, "r");
    if (file1 == NULL) {
        printf("Error: Could not open file %s\n", filename1);
        return 1;
    }

    file2 = fopen(filename2, "r");
    if (file2 == NULL) {
        printf("Error: Could not open file %s\n", filename2);
        return 1;
    }

    mergedFile = fopen(mergedFilename, "w");
    if (mergedFile == NULL) {
        printf("Error: Could not create the merged file %s\n", mergedFilename);
        return 1;
    }

    // Copy contents of the first file to the merged file
    while ((ch = fgetc(file1)) != EOF) {
        fputc(ch, mergedFile);
    }

    // Copy contents of the second file to the merged file
    while ((ch = fgetc(file2)) != EOF) {
        fputc(ch, mergedFile);
    }

    printf("The files have been merged successfully into %s\n", mergedFilename);

    fclose(file1);
    fclose(file2);
    fclose(mergedFile);

    return 0;
}
