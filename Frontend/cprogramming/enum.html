Structure, Union and Enum in C

Structure (struct) in C

A structure is a user-defined data type that allows the grouping of variables of different types under a single name. Each variable in the structure is called a "member."

Syntax:

struct structure_name {
    data_type member1;
    data_type member2;
    // Add more members
};

Example:

#include <stdio.h>
struct Student {
    char name[50];
    int age;
    float marks;
};

int main() {
    struct Student student1 = {"John", 20, 85.5};

    printf("Name: %s\n", student1.name);
    printf("Age: %d\n", student1.age);
    printf("Marks: %.2f\n", student1.marks);

    return 0;
}

Advantages:

  • Grouping different types of data: Structures are useful for grouping different data types together, such as representing real-world objects (e.g., students, books, etc.).
  • Organized and readable: Structures make the code more readable and organized, as all related data can be accessed through one variable.

Disadvantages:

  • Memory usage: A structure allocates memory for all its members, meaning it can use more memory than needed when some members aren't used.

Union (union) in C

A union is similar to a structure, but with one key difference: all members of a union share the same memory location. This means that a union can only store one of its members at a time, and the memory size of a union is determined by the size of its largest member.

Syntax:

union union_name {
    data_type member1;
    data_type member2;
    // Add more members
};

Example:

#include <stdio.h>
union Data {
    int i;
    float f;
    char str[20];
};

int main() {
    union Data data1;
    
    data1.i = 10;
    printf("Integer: %d\n", data1.i);
    
    data1.f = 3.14;
    printf("Float: %.2f\n", data1.f);
    
    // After assigning `f`, `i` value gets overwritten
    data1.i = 100;
    printf("Integer after overwrite: %d\n", data1.i);

    return 0;
}

Advantages:

  • Memory efficiency: Unions are memory-efficient, as they use the memory only for one of the members at a time. This can save memory, especially when the union has large data types.

Disadvantages:

  • Only one member at a time: Only one of the union members can store data at any given time. This makes unions less flexible than structures when you need to store multiple values simultaneously.

Enum (enum) in C

An enum is a data type that allows you to assign names to integral constants. It is typically used to represent a set of related values.

Syntax:

enum enum_name {
    constant1 = value1,
    constant2 = value2,
    // Add more constants
};

Example:

#include <stdio.h>
enum Weekday {
    Sunday = 1, 
    Monday, 
    Tuesday, 
    Wednesday, 
    Thursday, 
    Friday, 
    Saturday
};

int main() {
    enum Weekday today = Wednesday;

    printf("Today is day number %d of the week.\n", today);  // Output: Today is day number 4 of the week.
    return 0;
}

Advantages:

  • Readability: Using enum helps make code more readable by providing meaningful names to constant values, which improves the maintainability of the code.
  • Better control over values: You can explicitly assign values to the constants, or let the compiler automatically assign them in sequence.

Disadvantages:

  • Limited to integers: enum values are treated as integers, so you cannot use non-integer values.

Comparison of Structure, Union, and Enum

Feature Structure Union Enum
Memory Allocation Allocates memory for all members. Allocates memory for the largest member only. No memory allocated, it's a symbolic name for a constant.
Use Case Useful for grouping different data types together. Useful when you need to store one of several data types, but not all at once. Used to represent a set of related integer constants.
Access to Members All members can be accessed independently. Only one member can be accessed at a time. Represents integer values using names (constants).
Memory Efficiency Less memory-efficient, as all members are stored. More memory-efficient as only one member is stored. No memory usage, only symbolic constants.
Flexibility Very flexible. Can store multiple different types of data. Less flexible as only one member can be used at any time. Fixed values and cannot store data or objects.
Size Size is the sum of the sizes of its members. Size is the size of its largest member. No size, as it only represents integer constants.

1.Write a program to define a structure to store details of a student (name, roll number, and marks in three subjects). Use this structure to input and display the details of three students.

#include <stdio.h>
struct Student {
    char name[50];
    int roll_no;
    float marks[3];
};

int main() {
    struct Student students[3];
    for(int i = 0; i < 3; i++) {
        printf("Enter details for student %d:\n", i + 1);
        printf("Name: ");
        scanf("%s", students[i].name);
        printf("Roll Number: ");
        scanf("%d", &students[i].roll_no);
        printf("Enter marks for three subjects: ");
        for(int j = 0; j < 3; j++) {
            scanf("%f", &students[i].marks[j]);
        }
    }

    printf("\nStudent Details:\n");
    for(int i = 0; i < 3; i++) {
        printf("Student %d\n", i + 1);
        printf("Name: %s\n", students[i].name);
        printf("Roll Number: %d\n", students[i].roll_no);
        printf("Marks: %.2f, %.2f, %.2f\n", students[i].marks[0], students[i].marks[1], students[i].marks[2]);
    }

    return 0;
}

2.Write a program to define a structure to represent a date (day, month, year). Write functions to input, display, and calculate the difference between two dates.

#include <stdio.h>
struct Date {
    int day;
    int month;
    int year;
};

void inputDate(struct Date* date) {
    printf("Enter day: ");
    scanf("%d", &date->day);
    printf("Enter month: ");
    scanf("%d", &date->month);
    printf("Enter year: ");
    scanf("%d", &date->year);
}

void displayDate(struct Date date) {
    printf("%d/%d/%d\n", date.day, date.month, date.year);
}

int isLeapYear(int year) {
    if ((year % 4 == 0 && year % 100 != 0) || (year % 400 == 0)) {
        return 1;
    }
    return 0;
}

int daysInMonth(int month, int year) {
    switch (month) {
        case 1: case 3: case 5: case 7: case 8: case 10: case 12: return 31;
        case 4: case 6: case 9: case 11: return 30;
        case 2: return isLeapYear(year) ? 29 : 28;
        default: return 0;
    }
}

int dateToDays(struct Date date) {
    int totalDays = 0;
    for (int i = 1; i < date.year; i++) {
        totalDays += isLeapYear(i) ? 366 : 365;
    }
    for (int i = 1; i < date.month; i++) {
        totalDays += daysInMonth(i, date.year);
    }
    totalDays += date.day;
    return totalDays;
}

void calculateDateDifference(struct Date date1, struct Date date2) {
    int days1 = dateToDays(date1);
    int days2 = dateToDays(date2);
    int diff = days2 - days1;
    if (diff < 0) {
        diff = -diff;
    }
    printf("The difference between the two dates is %d days.\n", diff);
}

int main() {
    struct Date date1, date2;
    printf("Enter first date:\n");
    inputDate(&date1);
    printf("Enter second date:\n");
    inputDate(&date2);

    printf("First date: ");
    displayDate(date1);

    printf("Second date: ");
    displayDate(date2);

    calculateDateDifference(date1, date2);

    return 0;
}

4.Write a program using a union to store a student's marks either as an integer (total marks) or as a float (CGPA). Include a function to display the marks in the appropriate format.

#include <stdio.h>
union Marks {
    int totalMarks;
    float CGPA;
};

struct Student {
    char name[50];
    int rollNo;
    union Marks marks;
    int isCGPA;  // Flag to indicate whether the marks are in CGPA format
};

void inputStudentDetails(struct Student* student) {
    printf("Enter student name: ");
    getchar();  // To consume any leftover newline character from previous input
    fgets(student->name, sizeof(student->name), stdin);
    
    printf("Enter roll number: ");
    scanf("%d", &student->rollNo);

    printf("Enter marks as:\n1. Total Marks (Integer)\n2. CGPA (Float)\nSelect an option: ");
    int option;
    scanf("%d", &option);

    if (option == 1) {
        printf("Enter total marks: ");
        scanf("%d", &student->marks.totalMarks);
        student->isCGPA = 0;
    } else if (option == 2) {
        printf("Enter CGPA: ");
        scanf("%f", &student->marks.CGPA);
        student->isCGPA = 1;
    } else {
        printf("Invalid option. Defaulting to Total Marks.\n");
        printf("Enter total marks: ");
        scanf("%d", &student->marks.totalMarks);
        student->isCGPA = 0;
    }
}

void displayStudentDetails(struct Student student) {
    printf("\nStudent Details:\n");
    printf("Name: %s", student.name);
    printf("Roll Number: %d\n", student.rollNo);
    
    if (student.isCGPA) {
        printf("Marks (CGPA): %.2f\n", student.marks.CGPA);
    } else {
        printf("Marks (Total): %d\n", student.marks.totalMarks);
    }
}

int main() {
    struct Student student;
    
    inputStudentDetails(&student);
    displayStudentDetails(student);
    
    return 0;
}

5.Define an enum for months of the year. Write a program that accepts a month number from the user and displays the number of days in that month, accounting for leap years.

#include <stdio.h>

enum Month {
    JANUARY = 1, FEBRUARY, MARCH, APRIL, MAY, JUNE, JULY, AUGUST, SEPTEMBER, OCTOBER, NOVEMBER, DECEMBER
};

int isLeapYear(int year) {
    if ((year % 4 == 0 && year % 100 != 0) || (year % 400 == 0)) {
        return 1;
    }
    return 0;
}

int daysInMonth(int month, int year) {
    switch (month) {
        case JANUARY: case MARCH: case MAY: case JULY: case AUGUST: case OCTOBER: case DECEMBER:
            return 31;
        case APRIL: case JUNE: case SEPTEMBER: case NOVEMBER:
            return 30;
        case FEBRUARY:
            return isLeapYear(year) ? 29 : 28;
        default:
            return 0;
    }
}

int main() {
    int month, year;
    
    printf("Enter month number (1-12): ");
    scanf("%d", &month);
    
    if (month < 1 || month > 12) {
        printf("Invalid month number.\n");
        return 1;
    }
    
    printf("Enter year: ");
    scanf("%d", &year);
    
    int days = daysInMonth(month, year);
    printf("Number of days in month %d of year %d: %d\n", month, year, days);
    
    return 0;
}
