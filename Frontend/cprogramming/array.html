Arrays in C Programming

An array in C is a collection of variables of the same data type stored in contiguous memory locations. It allows you to store multiple values under one variable name, making it easier to manage and access data in your program.

Key Points to Understand About Arrays:

  • Array Declaration: Defines the array and its data type.
  • Array Initialization: Assigns values to the array elements.
  • Accessing Array Elements: Uses the index to access or modify array values.
  • Array Size: The number of elements an array can hold.
  • Multidimensional Arrays: Arrays with more than one dimension, like matrices.

1. Declaring Arrays

In C, you declare an array by specifying the data type, the name of the array, and the number of elements it can hold.

Syntax:

type array_name[array_size];
  • type: The data type of the elements (e.g., int, float, char).
  • array_name: The name of the array.
  • array_size: The number of elements the array can store.

Example:

#include <stdio.h>
int main() {
    int arr[5];  // Declares an integer array of size 5

    // Assigning values to the array
    arr[0] = 10;
    arr[1] = 20;
    arr[2] = 30;
    arr[3] = 40;
    arr[4] = 50;

    // Printing the elements of the array
    for (int i = 0; i < 5; i++) {
        printf("Element at index %d: %d\\n", i, arr[i]);
    }

    return 0;
}

2. Array Initialization

Arrays can be initialized at the time of declaration, which allows you to assign values to the array elements directly.

Syntax:

type array_name[] = {value1, value2, value3, ...};

Example:

#include <stdio.h>
int main() {
    int arr[] = {10, 20, 30, 40, 50};  // Array initialization

    // Printing the elements of the array
    for (int i = 0; i < 5; i++) {
        printf("Element at index %d: %d\\n", i, arr[i]);
    }
    return 0;
}

3. Accessing Array Elements

Array elements can be accessed using indices. C arrays are 0-indexed, meaning the first element is at index 0, the second element at index 1, and so on.

Syntax:

array_name[index]

Example:

#include <stdio.h>
int main() {
    int arr[] = {10, 20, 30, 40, 50};

    // Accessing elements using index
    printf("First element: %d\\n", arr[0]);  // Output: 10
    printf("Second element: %d\\n", arr[1]); // Output: 20
    printf("Last element: %d\\n", arr[4]);   // Output: 50

    return 0;
}

4. Multidimensional Arrays

A multidimensional array is an array of arrays. It can be used to represent matrices or tables. C supports arrays with two or more dimensions.

Syntax for 2D Array:

type array_name[row_size][column_size];

Example:

#include <stdio.h>
int main() {
    int arr[2][3] = { {1, 2, 3}, {4, 5, 6} };

    // Accessing elements of a 2D array
    for (int i = 0; i < 2; i++) {
        for (int j = 0; j < 3; j++) {
            printf("Element at arr[%d][%d]: %d\\n", i, j, arr[i][j]);
        }
    }  
    return 0;
}

5. Passing Arrays to Functions

You can pass arrays to functions in C by passing the address of the first element. Inside the function, the array can be accessed by using the array name or a pointer.

Example:

#include <stdio.h>
void printArray(int arr[], int size) {
    for (int i = 0; i < size; i++) {
        printf("Element at index %d: %d\\n", i, arr[i]);
    }
}
int main() {
    int arr[] = {10, 20, 30, 40, 50};

    // Passing array to function
    printArray(arr, 5); 
    return 0;
}

6. Array Size

The size of an array is determined at compile time. In C, you can calculate the size of an array using the sizeof operator.

Example:

#include <stdio.h>
int main() {
    int arr[] = {10, 20, 30, 40, 50};

    // Calculating size of the array
    int size = sizeof(arr) / sizeof(arr[0]);

    printf("Size of the array: %d\\n", size);

    return 0;
}

5.Write a program to find the second largest element in an array.

#include <stdio.h>
int main() {
    int n, i, largest, secondLargest;
    scanf("%d", &n);
    int arr[n];
    
    for(i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }
    
    largest = secondLargest = -2147483648;
    
    for(i = 0; i < n; i++) {
        if(arr[i] > largest) {
            secondLargest = largest;
            largest = arr[i];
        } else if(arr[i] > secondLargest && arr[i] != largest) {
            secondLargest = arr[i];

    if(secondLargest == -2147483648) {
        printf("No second largest element");
    } else {

7.Write a program to merge two sorted arrays into a single sorted array.

#include <stdio.h>
int main() {
    int n1, n2, i, j, k;
    scanf("%d", &n1);
    int arr1[n1];
    
    for(i = 0; i < n1; i++) {
        scanf("%d", &arr1[i]);
    }

    scanf("%d", &n2);
    int arr2[n2];
    
    for(i = 0; i < n2; i++) {
        scanf("%d", &arr2[i]);
    }

    int merged[n1 + n2];
    i = 0; j = 0; k = 0;
    
    while(i < n1 && j < n2) {
        if(arr1[i] <= arr2[j]) {
            merged[k++] = arr1[i++];
        } else {
            merged[k++] = arr2[j++];
        }
    }

    while(i < n1) {
        merged[k++] = arr1[i++];
    }

    while(j < n2) {
        merged[k++] = arr2[j++];
    }

    for(i = 0; i < n1 + n2; i++) {
        printf("%d ", merged[i]);
    }
    
    return 0;
}

8.Write a program to rotate an array by n positions.

#include <stdio.h>
void rotateArray(int arr[], int n, int d) {
    int temp[d], i;

    for(i = 0; i < d; i++) {
        temp[i] = arr[i];
    }

    for(i = d; i < n; i++) {
        arr[i - d] = arr[i];
    }

    for(i = 0; i < d; i++) {
        arr[n - d + i] = temp[i];
    }
}

int main() {
    int n, d, i;
    scanf("%d", &n);
    int arr[n];

    for(i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    scanf("%d", &d);

    rotateArray(arr, n, d);

    for(i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }

    return 0;
}

10.Write a program to multiply two matrices using a 2D array.

#include <stdio.h>
int main() {
    int m1, n1, m2, n2, i, j, k;

    scanf("%d %d", &m1, &n1);
    int matrix1[m1][n1];

    for(i = 0; i < m1; i++) {
        for(j = 0; j < n1; j++) {
            scanf("%d", &matrix1[i][j]);
        }
    }

    scanf("%d %d", &m2, &n2);
    int matrix2[m2][n2];

    for(i = 0; i < m2; i++) {
        for(j = 0; j < n2; j++) {
            scanf("%d", &matrix2[i][j]);
        }
    }

    if(n1 != m2) {
        printf("Matrix multiplication is not possible.");
        return 0;
    }

    int result[m1][n2];

    for(i = 0; i < m1; i++) {
        for(j = 0; j < n2; j++) {
            result[i][j] = 0;
            for(k = 0; k < n1; k++) {
                result[i][j] += matrix1[i][k] * matrix2[k][j];
            }
        }
    }

    for(i = 0; i < m1; i++) {
        for(j = 0; j < n2; j++) {
            printf("%d ", result[i][j]);
        }
        printf("\n");
    }

    return 0;
}
