1. Sum of Elements in an Array

2. Reverse an Array

Problem: Write a program to reverse an array.

#include <stdio.h>
void reverseArray(int arr[], int size) {
    int temp, start = 0, end = size - 1;
    while (start < end) {
        temp = arr[start];
        arr[start] = arr[end];
        arr[end] = temp;
        start++;
        end--;
    }
}

int main() {
    int arr[] = {1, 2, 3, 4, 5};
    int size = sizeof(arr) / sizeof(arr[0]);
    reverseArray(arr, size);
    for (int i = 0; i < size; i++) {
        printf("%d ", arr[i]);
    }
    return 0;
}

Output: 5 4 3 2 1

4. Palindrome Check (String)

Problem: Check if a string is a palindrome.

#include <stdio.h>
#include <string.h>
int isPalindrome(char str[]) {
    int start = 0, end = strlen(str) - 1;
    while (start < end) {
        if (str[start] != str[end]) {
            return 0;
        }
        start++;
        end--;
    }
    return 1;
}

int main() {
    char str[] = "madam";
    if (isPalindrome(str)) {
        printf("Palindrome\n");
    } else {
        printf("Not Palindrome\n");
    }
    return 0;
}

5. Count Vowels in a String

Problem: Write a program to count the number of vowels in a string.

#include <stdio.h>
#include <string.h>
int countVowels(char str[]) {
    int count = 0;
    for (int i = 0; i < strlen(str); i++) {
        if (str[i] == 'a' || str[i] == 'e' || str[i] == 'i' || str[i] == 'o' || str[i] == 'u') {
            count++;
        }
    }
    return count;
}

int main() {
    char str[] = "hello world";
    printf("Vowels Count: %d\n", countVowels(str));

}

6. Count the Number of Digits in a Number

Problem: Count the number of digits in a given number.

#include <stdio.h>
int countDigits(int num) {
    int count = 0;
    while (num != 0) {
        num /= 10;
        count++;
    }
    return count;
}

int main() {

7. Fibonacci Series (Recursive)

Problem: Print the Fibonacci series using recursion.

#include <stdio.h>
int fibonacci(int n) {
    if (n <= 1) return n;
    return fibonacci(n-1) + fibonacci(n-2);
}

int main() {

8. Factorial of a Number

Problem: Write a program to find the factorial of a number.

#include <stdio.h>
int factorial(int n) {
    if (n == 0 || n == 1) return 1;
    return n * factorial(n - 1);
}

int main() {
    int n = 5;
    printf("Factorial: %d\n", factorial(n));

11. Prime Number Check

Problem: Write a program to check if a number is prime.

#include <stdio.h>
int isPrime(int num) {
    for (int i = 2; i <= num / 2; i++) {
        if (num % i == 0) return 0;
    }
    return 1;
}

int main() {
    int num = 11;
    if (isPrime(num)) {
        printf("%d is prime\n", num);
    } else {
        printf("%d is not prime\n", num);
    }
    return 0;
}
